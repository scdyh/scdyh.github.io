我的逆向水品是在比较一般，丢给ai了，大概就是将输入字符与 RC4 生成的密钥流异或后，再与内存中的目标字节进行比较。

```python3
def rc4_init(key: bytes):
    s = list(range(256))
    j = 0
    for i in range(256):
        j = (j + s[i] + key[i % len(key)]) & 0xff
        s[i], s[j] = s[j], s[i]
    return s

def rc4_keystream(s, n):
    i = 0
    j = 0
    out = []
    for _ in range(n):
        i = (i + 1) & 0xff
        j = (j + s[i]) & 0xff
        s[i], s[j] = s[j], s[i]
        out.append(s[(s[i] + s[j]) & 0xff])
    return out

key = b"welcome"
cipher = bytes.fromhex("1de12092c435d78c55915efa788eb296a047")

s = rc4_init(key)
ks = rc4_keystream(s, len(cipher))
plain = bytes([c ^ k for c, k in zip(cipher, ks)])

print(plain)
```
